\documentclass{experimento}

\edef\nRows{8}

\begin{document}

\IgnoreSpacesOn
\prgNewFunction \generarTabla {} {
  \tlNew \lTablaTl
  \tlSet \lTablaTl {a & b \\}
  \tlPutRight \lTablaTl {c & d\\}
  \intStepInline {1} {1} {10} {
    \tlPutRight \lTablaTl {e & f\\}
  }
\prgReturn {\tlUse \lTablaTl}
}
\IgnoreSpacesOff

\begin{tblr}[evaluate=\generarTabla]{hlines}
  Row1 & 1 \\
  Row3 & 3 \\
  Row5 & 5 \\
  \generarTabla
\end{tblr}


\IgnoreSpacesOn
\tlNew \lPlotTl

\prgNewFunction \puntos {} {
  \tlClear \lPlotTl
  \tlPutRight \lPlotTl {(1,2)}
  \tlPutRight \lPlotTl {(2,5)}
  \prgReturn {\expValue \lPlotTl}
}
% First store the result
\tlNew \lPuntosTl
\tlSet \lPuntosTl {\expWhole {\puntos}}
\edef\coords{\lPuntosTl}

\IgnoreSpacesOff

\begin{tikzpicture}
  \begin{axis}
    \addplot[] coordinates {\coords};
  \end{axis}
\end{tikzpicture}

\end{document}

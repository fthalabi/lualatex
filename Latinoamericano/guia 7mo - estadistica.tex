\begin{luacode}
  require('estadistica.lua')
  function lista_a_string(lista)
    local str = ""
    for _, v in pairs(lista) do
      str = str .. string.format("%d \\\\ ", v)
    end
    return str
  end
  function a_puntos(x,y)
    local str = ""
    for i, vx in pairs(x) do
      str = str ..string.format(" (%d,%d) ", vx, y[i])
    end
    return str
  end
\end{luacode}
